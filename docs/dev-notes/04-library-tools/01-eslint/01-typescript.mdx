## Configure

<ConfigTip />

Install the following plugins and configs:

```terminal
pnpm add --save-dev @eslint/compat @eslint/js @eslint/css @eslint/json @eslint/markdown typescript-eslint
```

Modify the `eslint.config.mjs` file to match the following:

```ts title="eslint.config.mjs"
import { fileURLToPath } from "node:url";
import { defineConfig } from "eslint/config";
import { includeIgnoreFile } from "@eslint/compat";
import js from "@eslint/js";
// import css from "@eslint/css"; // 游리 TODO: Does this work yet for TailwindCSS v4.0?
import json from "@eslint/json";
import markdown from "@eslint/markdown";
import tseslint from "typescript-eslint";

const gitignorePath = fileURLToPath(new URL(".gitignore", import.meta.url));

export default defineConfig([
  includeIgnoreFile(gitignorePath),
  {
    files: ["**/*.{js,jsx,ts,tsx,mjs,cjs,mts,cts}"],
    extends: [js.configs.recommended, tseslint.configs.recommended],
  },
  // 游리 TODO: If you are NOT using TailwindCSS then it should be safe to uncomment this object and
  // remove the `languageOptions` property
  // 游리 TODO: Does this work yet for TailwindCSS v4.0?
  // {
  //   files: ["**/*.css"],
  //   language: "css/css",
  //   languageOptions: {
  //     tolerant: true, // 游리 This may be required due to custom syntax that TailwindCSS uses
  //     customSyntax: tailwindSyntax, // 游리 This has not been updated yet for TailwindCSS v4.0
  //   },
  //   extends: [css.configs.recommended],
  // },
  {
    files: ["**/*.json"],
    ignores: ["package-lock.json"],
    language: "json/json",
    extends: [json.configs.recommended],
  },
  {
    files: ["**/*.md"],
    extends: [markdown.configs.recommended],
    language: "markdown/gfm", // 游리 Optional, include this property if using Github-flavored markdown
    rules: {
      "markdown/no-missing-label-refs": "off", // 游리 Optional, this rule doesn't work with Github-flavored markdown alerts
    },
  },
]);
```
